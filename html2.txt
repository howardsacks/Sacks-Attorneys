    <script>
        // Updated JavaScript for crawl-friendly navigation
        document.addEventListener('DOMContentLoaded', function() {
            // Handle anchor clicks
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    const targetId = this.getAttribute('href');
                    if (targetId === '#') return;
                    
                    const targetElement = document.querySelector(targetId);
                    if (targetElement) {
                        // Hide all sections (visually)
                        document.querySelectorAll('.page-section').forEach(section => {
                            section.classList.remove('active');
                            section.style.position = 'absolute';
                            section.style.left = '-9999px';
                            section.style.top = '-9999px';
                        });
                        
                        // Show target section
                        targetElement.classList.add('active');
                        targetElement.style.position = 'static';
                        targetElement.style.left = 'auto';
                        targetElement.style.top = 'auto';
                        
                        // Update URL without page reload
                        if (history.pushState) {
                            history.pushState(null, '', targetId === '#home' ? '/' : targetId);
                        }
                        
                        // Update active nav link
                        document.querySelectorAll('nav a').forEach(link => {
                            link.classList.remove('active');
                        });
                        this.classList.add('active');
                        
                        // Close mobile menu if open
                        const menu = document.getElementById('mainMenu');
                        if (menu && menu.classList.contains('active')) {
                            menu.classList.remove('active');
                            document.getElementById('mobileMenuBtn').querySelector('i').className = 'fas fa-bars';
                        }
                        
                        // Smooth scroll to top
                        window.scrollTo({ top: 0, behavior: 'smooth' });
                    }
                });
            });
            
            // Mobile menu toggle
            document.getElementById('mobileMenuBtn').addEventListener('click', function() {
                const menu = document.getElementById('mainMenu');
                menu.classList.toggle('active');
                const icon = this.querySelector('i');
                icon.className = menu.classList.contains('active') ? 'fas fa-times' : 'fas fa-bars';
            });
            
            // Handle initial load
            const hash = window.location.hash || '#home';
            const targetElement = document.querySelector(hash);
            if (targetElement) {
                document.querySelectorAll('.page-section').forEach(section => {
                    section.classList.remove('active');
                    section.style.position = 'absolute';
                    section.style.left = '-9999px';
                    section.style.top = '-9999px';
                });
                targetElement.classList.add('active');
                targetElement.style.position = 'static';
                targetElement.style.left = 'auto';
                targetElement.style.top = 'auto';
                
                // Set active nav link
                document.querySelectorAll('nav a').forEach(link => {
                    link.classList.remove('active');
                    if (link.getAttribute('href') === hash) {
                        link.classList.add('active');
                    }
                });
            }
        });
    </script>
